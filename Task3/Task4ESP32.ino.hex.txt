#include <DHT.h>

#define DHTPIN 4
#define DHTTYPE DHT11
#define LED_PIN 2

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
  pinMode(LED_PIN, OUTPUT);
  Serial.println("Dashboard Started!");
}

void loop() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  Serial.print("Temp: ");
  Serial.print(t);
  Serial.print("C | Humidity: ");
  Serial.print(h);
  Serial.println("%");
  delay(2000);
}
